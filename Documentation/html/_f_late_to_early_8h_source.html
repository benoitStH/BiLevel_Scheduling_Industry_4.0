108 void swapLateToEarly(SwapOperation& swapOp, const Machine& m1, const Machine& m2, unsigned int k, float speed)
109 {
110 const Job& job1 = m1.getAffectedJob()[k];
111 const Job& job2 = m2.getAffectedJob()[k];
112
113 float p1 = job1.getPi() / speed;
114 float p2 = job2.getPi() / speed;
115 float w1 = job1.getWi();
116 float w2 = job2.getWi();
117 float d1 = job1.getDi();
118 float d2 = job2.getDi();
119
120 bool swapIsPossible; // Is a swap possible ?
121
122 // Are the jobs both late and swapping will make job1 early ?
123 swapIsPossible = (job1.isLate() && job2.isLate() && (m2.startTimeOfJob(k) + p1) < d1);
124
125 // Otherwise, Is job1 the only one late ?
126 if (job1.isLate() && !job2.isLate() && swapIsPossible == false)
127 {
128 // Swapping will make job1 early and job2 will still be early
129 swapIsPossible = ((m2.startTimeOfJob(k) + p1 < d1) &&
130 m1.startTimeOfJob(k) + p2 < d2);
131 if (swapIsPossible) {
132 swapOp.gain = w1;
133 }
134
135 // Or Swapping will make job1 early and job2 will be late, Swap is possible if w1 > w2
136 swapIsPossible = (m2.startTimeOfJob(k) + p1 < d1);
137 if (swapIsPossible && w1 > w2)
138 {
139 swapOp.gain = w1 - w2;
140 }
141
142 }
143 else if (job2.isLate() && !job1.isLate()) // Is job2 the only one late ?
144 {
145 // Swapping will make job2 early and job1 will still be early
146 swapIsPossible = ((m2.startTimeOfJob(k) + p1 < d1) &&
147 m1.startTimeOfJob(k) + p2 < d2);
148 if (swapIsPossible)
149 {
150 swapOp.gain = w2;
151 }
152
153 // Or Swapping will make job2 early and job1 will be late, Swap is possible if w2 > w1
154 swapIsPossible = (m1.startTimeOfJob(k) + p2 < d2);
155 if (swapIsPossible && w2 > w1)
156 {
157 swapOp.gain = w2 - w1;
158 }
159 }

191 void swapLateness(SwapOperation& swapOp, const Solution& s)
192 {
193 unsigned int k = swapOp.bloc;
194 const Machine& m1 = s.getMachine(swapOp.machine1);
195 const Machine& m2 = s.getMachine(swapOp.machine2);
196
197 const Job& job1 = m1.getAffectedJob()[k];
198 const Job& job2 = m2.getAffectedJob()[k];
199 float speed = m1.getSpeed();
200
201 float pi_diff = (job1.getPi() - job2.getPi()) / speed;
202
203 // If it is the last bloc, use the dumb method
204 if (swapOp.bloc == m1.getAffectedJob().size() - 1)
205 {
206 swapDumbly(swapOp, s);
207 return;
208 }
209
210
211 float lateness1; // lateness of job1's following job
212 do
213 {
214 const Job& nextJob1 = m1.getAffectedJob()[k + 1]; // job1's following job
215 lateness1 = m1.startTimeOfJob(swapOp.bloc + 2) - float(nextJob1.getDi()); // computing its lateness
216
217 // If it is late and the lateness is smaller than the difference of job1's and job2's processing time
218 if (0 < lateness1 && lateness1 <= pi_diff)
219 {
220 // Initialize the swap's gain at 0 if that is not the case already
221 if (swapOp.gain < 0) { swapOp.gain = 0; }
222
223 // If we choose this swap, this following job will become early, we add its weight to our swap's gain
224 swapOp.gain += nextJob1.getWi();
225 }
226
227 k++; // We check the next following job until we reach the one before last one
228 } while (k < m1.getAffectedJob().size() - 1);
229
230
231 }

239 void swapLatenessCareful(SwapOperation& swapOp, const Solution& s)
240 {
241 unsigned int k = swapOp.bloc;
242 const Machine& m1 = s.getMachine(swapOp.machine1);
243 const Machine& m2 = s.getMachine(swapOp.machine2);
244
245 const Job& job1 = m1.getAffectedJob()[k];
246 const Job& job2 = m2.getAffectedJob()[k];
247 float speed = m1.getSpeed();
248
249 // If it is the last bloc, use the dumb method
250 if (swapOp.bloc == m1.getAffectedJob().size() - 1)
251 {
252 swapDumbly(swapOp, s);
253 return;
254 }
255
256 swapOp.gain = 0;
257
258 // If job1 is late and will be early after swapping
259 if (job1.isLate() && m2.startTimeOfJob(k) + (job1.getPi() / speed) <= job1.getDi())
260 {
261 swapOp.gain += job1.getWi();
262 }
263
264 // If job1 is early and will be late after swapping
265 if (job1.isLate() == false && m2.startTimeOfJob(k) + (job1.getPi() / speed) > job1.getDi())
266 {
267 swapOp.gain -= job1.getWi();
268 }
269
270 // If job2 is late and will be early after swapping
271 if (job2.isLate() && m2.startTimeOfJob(k) + (job2.getPi() / speed) <= job2.getDi())
272 {
273 swapOp.gain += job2.getWi();
274 }
275
276 // If job2 is early and will be late after swapping
277 if (job2.isLate() == false && m2.startTimeOfJob(k) + (job2.getPi() / speed) > job2.getDi())
278 {
279 swapOp.gain -= job2.getWi();
280 }
281
282 // Check job1's following jobs
283 swapLateness(swapOp, s);
284 }

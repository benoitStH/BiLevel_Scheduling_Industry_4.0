64 float startTimeOfJob(unsigned int j) const {
65 if (j == 0) { return 0; }
66
67 return startTimeOfJob(j - 1) + listAffectedJobs[j-1].getPi() / speed;
68 }

74 void evaluate() {
75 float cumulativeWeightedTardyJob = 0.0; float cumulativeCompletionTimes = 0.0; float completionTimes = 0.0;
76 size_t nbJobs = listAffectedJobs.size();
77 for (size_t i = 0; i < nbJobs; i++) {
78 // compute cumulative completion times, a job at position i count (nbJobs-i) times
79 cumulativeCompletionTimes += (nbJobs - i) * listAffectedJobs[i].getPi() / speed;
80 completionTimes += listAffectedJobs[i].getPi() / speed;
81
82 // the job is late if the completion time is past its due date
83 listAffectedJobs[i].setLate((listAffectedJobs[i].getDi() < completionTimes));
84 cumulativeWeightedTardyJob += listAffectedJobs[i].isLate() ? listAffectedJobs[i].getWi() : 0.0;
85 }
86 sum_Cj = cumulativeCompletionTimes;
87 sum_wj_Uj = cumulativeWeightedTardyJob;
88 }

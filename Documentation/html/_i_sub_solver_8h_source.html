83 std::vector<Job> listLPT_jobs = listJobs;
84 std::sort(listLPT_jobs.begin(), listLPT_jobs.end(), Job::LPT_inv_EDD);

86 // total number of machines

88
89 // affectation weights of each machine which are each equal to (number of affected jobs + 1)/speed
90 std::vector<float> affectationWeights = std::vector<float>(nbMachines, float(1));
91
92 for (int i = 0; i < nbMachines; i++)
93 {
94 affectationWeights[i] = affectationWeights[i] / (*solution)[i].getSpeed();
95 }
96
97 // machine whose affectation weight is the smallest (only the first occurence)
98 unsigned int min_index;
99
100 // for each job
101 for (Job& job : listLPT_jobs)
102 {
103 // find the machine with smallest affectation weight
104 min_index = std::distance(affectationWeights.begin(), std::min_element(affectationWeights.begin(), affectationWeights.end()));
105
106 // inserting the job at the start of the machine's sequence
107 (*solution)[min_index].add_job(0, job);
108
109 // updating the machine's affectation weight
110 affectationWeights[min_index] = ((*solution)[min_index].getAffectedJob().size() + 1) / (*solution)[min_index].getSpeed();
111 }
112
113 unsigned int nbJobsOnHighSpeedMachines = solution->getMaxNumberOfHighSpeedBlocs();
114 unsigned int nbHighMachines = instance.getNbOfHighSpeedMachines();
115
116 unsigned int nbJobsOnLowSpeedMachines = solution->getMaxNumberOfLowSpeedBlocs();
117 unsigned int nbLowSpeedMachines = instance.getNbOfLowSpeedMachines();
118
119 // Adding ghost jobs (whose number, processing time, due date and weight are 0)
120 for (unsigned int m = 0; m < nbMachines; m++)
121 {
122 while ((*solution)[m].getAffectedJob().size() < ( m < nbHighMachines ? nbJobsOnHighSpeedMachines : nbJobsOnLowSpeedMachines))
123 {
124 // inserting a ghost job at the start of the machine's sequence
125 (*solution)[m].add_job(0, Job());

127 }
